Scale a 3x3 matrix

    2.0, 5.1, 3.0,
    1.0, -5.0, -1.2);
assert(a.scale(5.0) == Matrix3x3(5.0, 15.0, -20.0,
        10.0, 25.5, 15.0,
        5.0, -25.0, -6.0));
